184  subroutine shear_stress()
185  !-------------------------------------------------------------------------------------------------------------------------------
186  implicit none
187  real(R_P):: vol
188  type(type_tensor):: rst
189  !-------------------------------------------------------------------------------------------------------------------------------
190 
191  !-------------------------------------------------------------------------------------------------------------------------------
192  ! computing the value of the finite volume
193  select case(dir)
194  case('i')
195  vol=0.5_r_p*(block%C(i,j,k)%V + block%C(i+1,j,k)%V)
196  case('j')
197  vol=0.5_r_p*(block%C(i,j,k)%V + block%C(i,j+1,k)%V)
198  case('k')
199  vol=0.5_r_p*(block%C(i,j,k)%V + block%C(i,j,k+1)%V)
200  endselect
201  ! computing the gradient of velocity vector
202  rst%x%x = dudi_fv(u=vel%x,nsi=nfs%x,v=vol)
203  rst%x%y = dudi_fv(u=vel%x,nsi=nfs%y,v=vol)
204  rst%x%z = dudi_fv(u=vel%x,nsi=nfs%z,v=vol)
205  rst%y%x = dudi_fv(u=vel%y,nsi=nfs%x,v=vol)
206  rst%y%y = dudi_fv(u=vel%y,nsi=nfs%y,v=vol)
207  rst%y%z = dudi_fv(u=vel%y,nsi=nfs%z,v=vol)
208  rst%z%x = dudi_fv(u=vel%z,nsi=nfs%x,v=vol)
209  rst%z%y = dudi_fv(u=vel%z,nsi=nfs%y,v=vol)
210  rst%z%z = dudi_fv(u=vel%z,nsi=nfs%z,v=vol)
211  ! computing the shear stress tensor
212  tau%x%x = block%global%adim%Re_inv*rst%x%x*4._r_p/3._r_p
213  tau%x%y = block%global%adim%Re_inv*(rst%x%y+rst%y%x)
214  tau%x%z = block%global%adim%Re_inv*(rst%x%z+rst%z%x)
215  tau%y%x = tau%x%y
216  tau%y%y = block%global%adim%Re_inv*rst%y%y*4._r_p/3._r_p
217  tau%y%z = block%global%adim%Re_inv*(rst%y%z+rst%z%y)
218  tau%z%x = tau%x%z
219  tau%z%y = tau%y%z
220  tau%z%z = block%global%adim%Re_inv*rst%z%z*4._r_p/3._r_p
221  return
222  !-------------------------------------------------------------------------------------------------------------------------------
223  endsubroutine shear_stress
