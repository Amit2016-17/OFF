522  subroutine psendrecv(l,block)

528  integer(I_P):: proc
529  integer(I_P):: c

531  integer(I_P):: ierr,stat(mpi_status_size)

536  global => block(1)%global
537  do proc=0,nproc-1 ! the communications is organized by processes sequence
538  if ((ncr(proc,l)==0).AND.(ncr(proc,l)==0)) cycle ! there are no data to communicate to process proc
539  ! building the send buffer (Psend) of myrank for proc using local var P
540  do c=bbs(1,proc,l),bbs(2,proc,l)
541  b = minloc(array=blockmap,dim=1,mask=blockmap==sendmap(1,c))

543  enddo
544  ! sending Psend of myrank to proc and storing in Precv of proc
545  call mpi_sendrecv(psend(1,bbs(1,proc,l)),global%Np*ncs(proc,l),mpi_real8,proc,tagshift+nproc*(global%myrank+1), &
546  precv(1,bbr(1,proc,l)),global%Np*ncr(proc,l),mpi_real8,proc,tagshift+nproc*(proc +1), &
547  mpi_comm_world,stat,ierr)
548  ! coping the receive buffer (Precv) of myrank from proc in the local var P
549  do c=bbr(1,proc,l),bbr(2,proc,l)
550  call block(recvmap(1,c))%C(recvmap(2,c),recvmap(3,c),recvmap(4,c))%P%array2prim(precv(:,c))
551  enddo
552  enddo
553  return

592  function procmap_load(filename,global) result(err)
593  !---------------------------------------------------------------------------------------------------------------------------------
594  implicit none
595  character(*), intent(IN):: filename
596  type(type_global), intent(INOUT):: global
597  integer(I_P):: err
598  integer(I_P):: unitfree
599  logical:: is_file
600  integer(I_P):: b,bb
601  !---------------------------------------------------------------------------------------------------------------------------------
602 
603  !---------------------------------------------------------------------------------------------------------------------------------
604  inquire(file=trim(filename),exist=is_file,iostat=err)
605  if (.NOT.is_file) call file_not_found(global%myrank,filename,'procmap_load')
606  open(unit = get_unit(unitfree), file = trim(filename), status = 'OLD', action = 'READ', form = 'FORMATTED')
607  read(unitfree,*,iostat=err) global%Nb_tot
608  if (allocated(procmap)) deallocate(procmap) ; allocate(procmap(1:global%Nb_tot)) ; procmap = 0_i_p
609  read(unitfree,*,iostat=err)
610  do b=1,global%Nb_tot
611  read(unitfree,*,iostat=err) procmap(b) ! reading the process number of bth block
612  enddo
613  close(unitfree)
614  ! computing the local/global blocks map
615  if (nproc==1_i_p) then
616  ! there is no MPI environment thus all blocks are loaded by process 0
617  global%Nb = global%Nb_tot
618  if (allocated(blockmap)) deallocate(blockmap) ; allocate(blockmap(1:global%Nb)) ; blockmap = 0_i_p
619  do b=1,global%Nb_tot
620  blockmap(b) = b ! the blocks map is identity
621  enddo
622  else
623  ! computing the local (of myrank) number of blocks
624  global%Nb = 0
625  do b=1,global%Nb_tot
626  if (procmap(b)==global%myrank) global%Nb = global%Nb + 1
627  enddo
628  if (allocated(blockmap)) deallocate(blockmap) ; allocate(blockmap(1:global%Nb )) ; blockmap = 0_i_p
629  bb = 0
630  do b=1,global%Nb_tot
631  if (procmap(b)==global%myrank) then
632  bb = bb + 1
633  blockmap(bb) = b
634  endif
635  enddo
636  endif
637  return
638  !---------------------------------------------------------------------------------------------------------------------------------
639  endfunction procmap_load
