00497   !---------------------------------------------------------------------------------------------------------------------------------
00498   implicit none
00499   integer(I_P),      intent(IN)::    myrank                     !< Actual rank process.

00517       b = minloc(array=blockmap,dim=1,mask=blockmap==sendmap(1,c))

00525 #endif

00527     do c=bbR(1,proc,l),bbR(2,proc,l)
00528        block(recvmap(1,c))%fluid%P(recvmap(2,c),recvmap(3,c),recvmap(4,c)) = array2prim(Precv(:,c))
