125  subroutine init_amrblock(block,b)
126  !---------------------------------------------------------------------------------------------------------------------------------
127  implicit none
128  class(type_amrblock), intent(INOUT):: block
129  integer(I2P), intent(IN):: b
130  integer(I2P):: i,j,k
131  type(type_hashid):: id
132  type(type_cell):: cell
133  !---------------------------------------------------------------------------------------------------------------------------------
134 
135  !---------------------------------------------------------------------------------------------------------------------------------
136  ! initializing hash tables
137  call block%node%init
138  call block%Fi%init
139  call block%Fj%init
140  call block%Fk%init
141  call block%C%init
142  ! inserting data into the hash tables
143  do k=1-block%gc(5),block%Nk+block%gc(6)
144  do j=1-block%gc(3),block%Nj+block%gc(4)
145  do i=1-block%gc(1),block%Ni+block%gc(2)
146  call id%build(b=b,i=i,j=j,k=k,l=0_i1p,p=0_i8p)
147  call block%C%put(id,cell)
148  enddo
149  enddo
150  enddo
151  return
152  !---------------------------------------------------------------------------------------------------------------------------------
153  endsubroutine init_amrblock
