00071   elemental subroutine Rarefaction(sgn,g,delta,eta,u0,p0,a0,ux,rx,px,ax,s0,sx)
00072   !---------------------------------------------------------------------------------------------------------------------------------
00073   ! subroutine for computing an unknown state "x" from a known state "0" when the two states are separated by a rarefaction; it is
00074   ! assumed that the velocity of the unknown state "ux" is known. There is an input variable that indicates if the rarefaction
00075   ! propagates on the "u-a" direction (left) or on the "u+a" one (right).
00076   !---------------------------------------------------------------------------------------------------------------------------------
00077 
00078   !---------------------------------------------------------------------------------------------------------------------------------
00079   implicit none
00080   real(R_P),    intent(IN)::  sgn      ! sign for distinguishing "left" (-1) from "right" (1) wave
00081   real(R_P),    intent(IN)::  g        ! specific heats ratio
00082   real(R_P),    intent(IN)::  delta    ! (g-1)/2
00083   real(R_P),    intent(IN)::  eta      ! 2*g/(g-1)
00084   real(R_P),    intent(IN)::  u0,p0,a0 ! known state (speed, pressure and speed of sound)
00085   real(R_P),    intent(IN)::  ux       ! known speed of unknown state
00086   real(R_P),    intent(OUT):: rx,px,ax ! unknown pressure and density
00087   real(R_P),    intent(OUT):: s0,sx    ! wave speeds (head and back fronts)
00088   !---------------------------------------------------------------------------------------------------------------------------------
00089 
00090   !---------------------------------------------------------------------------------------------------------------------------------
00091   ax = a0 + sgn*delta*(ux - u0) ! unknown speed of sound
00092   px = p0*((ax/a0)**(eta))      ! unknown pressure
00093   rx = r(p=px,a=ax,g=g)         ! unknown density
00094   s0 = u0 + sgn*a0              ! left wave speed
00095   sx = ux + sgn*ax              ! right wave speed
00096   return

00114   real(R_P),    intent(OUT):: rx,px,ax ! unknown state (density, pressure and speed of sound).
00115   real(R_P),    intent(OUT):: ss       ! shock wave speed

00117   real(R_P)::                 x        ! dummy variable
00118   !---------------------------------------------------------------------------------------------------------------------------------
00119 
00120   !---------------------------------------------------------------------------------------------------------------------------------
00121   x   = 0.25_R_P*gp1*(ux - u0)/a0              ! dummy variable
