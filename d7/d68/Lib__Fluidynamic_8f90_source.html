178  subroutine compute_time(block,Dtmin)
179  !---------------------------------------------------------------------------------------------------------------------------------
180  implicit none
181  type(type_sblock), intent(INOUT):: block
182  real(R_P), intent(OUT):: dtmin
183  real(R_P):: vmax

185  real(R_P):: vmil,vmir,vmjl,vmjr,vmkl,vmkr
186  type(type_vector):: vm
187  integer(I_P):: i,j,k
188  !---------------------------------------------------------------------------------------------------------------------------------
189 
190  !---------------------------------------------------------------------------------------------------------------------------------
191  ! computing the minimum Dt into the inner cells
192  !$OMP PARALLEL DEFAULT(NONE) &
193  !$OMP PRIVATE(i,j,k,vmax,ss,vmiL,vmiR,vmjL,vmjR,vmkL,vmkR,vm) &
194  !$OMP SHARED(block,Dtmin)
195  !$OMP DO COLLAPSE(3)
196  do k=1,block%Nk
197  do j=1,block%Nj
198  do i=1,block%Ni
199  ! computing the local speed of sound
200  ss = a(p=block%C(i,j,k)%P%p,r=block%C(i,j,k)%P%d,g=block%C(i,j,k)%P%g)
201  ! evaluating the maximum propagation speed of acoustic segnals multiplied for face area
202  ! left i
203  vm = 0.5_r_p*(block%C(i-1,j,k)%P%v+block%C(i,j,k)%P%v)
204  vmil = (vm.dot.block%Fi(i-1,j,k)%N)*block%Fi(i-1,j,k)%S
205  vmil = abs(vmil) + ss
206  ! right i
207  vm = 0.5_r_p*(block%C(i,j,k)%P%v+block%C(i+1,j,k)%P%v)
208  vmir = (vm.dot.block%Fi(i,j,k)%N)*block%Fi(i,j,k)%S
209  vmir = abs(vmir) + ss
210  ! left j
211  vm = 0.5_r_p*(block%C(i,j-1,k)%P%v+block%C(i,j,k)%P%v)
212  vmjl = (vm.dot.block%Fj(i,j-1,k)%N)*block%Fj(i,j-1,k)%S
213  vmjl = abs(vmjl) + ss
214  ! right j
215  vm = 0.5_r_p*(block%C(i,j,k)%P%v+block%C(i,j+1,k)%P%v)
216  vmjr = (vm.dot.block%Fj(i,j,k)%N)*block%Fj(i,j,k)%S
217  vmjr = abs(vmjr) + ss
218  ! left k
219  vm = 0.5_r_p*(block%C(i,j,k-1)%P%v+block%C(i,j,k)%P%v)
220  vmkl = (vm.dot.block%Fk(i,j,k-1)%N)*block%Fk(i,j,k-1)%S
221  vmkl = abs(vmkl) + ss
222  ! right k
223  vm = 0.5_r_p*(block%C(i,j,k)%P%v+block%C(i,j,k+1)%P%v)
224  vmkr = (vm.dot.block%Fk(i,j,k)%N)*block%Fk(i,j,k)%S
225  vmkr = abs(vmkr) + ss
226  ! vmax
227  vmax = max(vmil,vmir,vmjl,vmjr,vmkl,vmkr)
228  block%C(i,j,k)%Dt = block%C(i,j,k)%V/vmax*block%global%CFL
229  enddo
230  enddo
231  enddo
232  ! computing minimum Dt
233  !$OMP SINGLE
234  dtmin = minval(block%C(1:block%Ni,1:block%Nj,1:block%Nk)%Dt)
235  !$OMP END SINGLE
236  ! ghost cells estrapolation: imposing the minum value of Dt
237  ! left i frame
238  !$OMP DO COLLAPSE(3)
239  do k=1-block%gc(5),block%Nk+block%gc(6)
240  do j=1-block%gc(3),block%Nj+block%gc(4)
241  do i=1-block%gc(1),0
242  block%C(i,j,k)%Dt = dtmin
243  enddo
244  enddo
245  enddo
246  ! right i frame
247  !$OMP DO COLLAPSE(3)
248  do k=1-block%gc(5),block%Nk+block%gc(6)
249  do j=1-block%gc(3),block%Nj+block%gc(4)
250  do i=block%Ni+1,block%Ni+block%gc(2)
251  block%C(i,j,k)%Dt = dtmin
252  enddo
253  enddo
254  enddo
255  ! left j frame
256  !$OMP DO COLLAPSE(3)
257  do k=1-block%gc(5),block%Nk+block%gc(6)
258  do j=1-block%gc(3),0
259  do i=1-block%gc(1),block%Ni+block%gc(2)
260  block%C(i,j,k)%Dt = dtmin
261  enddo
262  enddo
263  enddo
264  ! right j frame
265  !$OMP DO COLLAPSE(3)
266  do k=1-block%gc(5),block%Nk+block%gc(6)
267  do j=block%Nj+1,block%Nj+block%gc(4)
268  do i=1-block%gc(1),block%Ni+block%gc(2)
269  block%C(i,j,k)%Dt = dtmin
270  enddo
271  enddo
272  enddo
273  ! left k frame
274  !$OMP DO COLLAPSE(3)
275  do k=1-block%gc(5),0
276  do j=1-block%gc(3),block%Nj+block%gc(4)
277  do i=1-block%gc(1),block%Ni+block%gc(2)
278  block%C(i,j,k)%Dt = dtmin
279  enddo
280  enddo
281  enddo
282  ! right k frame
283  !$OMP DO COLLAPSE(3)
284  do k=block%Nk+1,block%Nk+block%gc(6)
285  do j=1-block%gc(3),block%Nj+block%gc(4)
286  do i=1-block%gc(1),block%Ni+block%gc(2)
287  block%C(i,j,k)%Dt = dtmin
288  enddo
289  enddo
290  enddo
291  !$OMP END PARALLEL
292  return

792  subroutine boundary_conditions(l,block)

877  !-------------------------------------------------------------------------------------------------------------------------------

898  enddo

905  ud%re = rk_stage(s1=s1,dt=dt,un=u%re ,ks=ksa(ns+4_i4p,1:s1-1_i1p))

907  return

909  endsubroutine rk_stages_sum_backend

914  subroutine rk_time_integration(block,RU)
915  !---------------------------------------------------------------------------------------------------------------------------------
916  implicit none
917  type(type_sblock), intent(INOUT):: block
918  real(R_P), intent(OUT):: ru(1:)
919  real(R_P) r(1:size(ru,dim=1))
920  integer(I4P):: i,j,k
921  !---------------------------------------------------------------------------------------------------------------------------------
922 
923  !---------------------------------------------------------------------------------------------------------------------------------
924  ru = 0._r_p
925  !$OMP PARALLEL DEFAULT(NONE) &
926  !$OMP PRIVATE(i,j,k,R) &
927  !$OMP SHARED(block) &
928  !$OMP REDUCTION(+: RU)
929  !$OMP DO COLLAPSE(3)
930  do k=1,block%Nk
931  do j=1,block%Nj
932  do i=1,block%Ni
933  call rk_time_integration_backend(ns=block%global%Ns,rk_ord=block%global%rk_ord,&
934  dt=block%C(i,j,k)%Dt,ks=block%C(i,j,k)%KS,u=block%C(i,j,k)%U,r=r)
935  ru = ru + (r*r)
936  enddo
937  enddo
938  enddo
939  !$OMP END PARALLEL
940  ru = sqrt(ru)
941  return
942  !---------------------------------------------------------------------------------------------------------------------------------
943  contains
944  pure subroutine rk_time_integration_backend(Ns,rk_ord,Dt,KS,U,R)
945  !-------------------------------------------------------------------------------------------------------------------------------
946  implicit none
947  integer(I_P), intent(IN):: ns
948  integer(I1P), intent(IN):: rk_ord
949  real(R_P), intent(IN):: dt
950  type(type_conservative), intent(IN):: ks(1:)
951  type(type_conservative), intent(INOUT):: u
952  real(R_P), intent(OUT):: r(1:)
953  real(R_P):: ksa(1:size(r,dim=1),1:rk_ord)
954  integer(I1P):: s1
955  integer(I4P):: s
956  !-------------------------------------------------------------------------------------------------------------------------------
957 
958  !-------------------------------------------------------------------------------------------------------------------------------
959  do s1=1_i1p,rk_ord
960  ksa(:,s1) = ks(s1)%cons2array()
961  enddo
962  do s=1_i4p,ns
963  call rk_time_integ(dt=dt,ks=ksa(s,:),r=r(s),unp1=u%rs(s))
964  enddo
965  call rk_time_integ(dt=dt,ks=ksa(ns+1_i4p,:),r=r(ns+1_i4p),unp1=u%rv%x)
966  call rk_time_integ(dt=dt,ks=ksa(ns+2_i4p,:),r=r(ns+2_i4p),unp1=u%rv%y)
967  call rk_time_integ(dt=dt,ks=ksa(ns+3_i4p,:),r=r(ns+3_i4p),unp1=u%rv%z)
968  call rk_time_integ(dt=dt,ks=ksa(ns+4_i4p,:),r=r(ns+4_i4p),unp1=u%re )
969  return
970  !-------------------------------------------------------------------------------------------------------------------------------
971  endsubroutine rk_time_integration_backend
972  endsubroutine rk_time_integration

976  subroutine solve_grl(l,block)
977  !---------------------------------------------------------------------------------------------------------------------------------
978  implicit none
979  integer(I_P), intent(IN):: l
980  type(type_sblock), intent(INOUT):: block(1:)
981  type(type_global), pointer:: global
982  real(R_P), allocatable:: dtmin(:)
983  real(R_P):: dtminl
984  real(R_P):: gdtmin
985  real(R_P), allocatable:: ru (:,:)
986  real(R_P), allocatable:: mru (:)
987  real(R_P), allocatable:: gmru(:)
988  integer(I_P):: err
989  integer(I_P):: b
990  integer(I1P):: s1
991  real(R_P):: sec_elp
992  real(R_P):: sec_res
993  type(type_time):: time_elp
994  type(type_time):: time_res
995  real(R_P):: progress
996  integer(I_P):: i,j,k
997  !---------------------------------------------------------------------------------------------------------------------------------
998 
999  !---------------------------------------------------------------------------------------------------------------------------------
1000  global => block(1)%global
1001  ! allocating dynamic arrays
1002  allocate(dtmin(1:global%Nb))
1003  allocate(ru(1:global%Nc,1:global%Nb))
1004  allocate(mru(1:global%Nc))
1005  allocate(gmru(1:global%Nc))
1006  ! converting conservative variables to primitive ones
1007 #ifdef PROFILING
1008  call profile(p=2,pstart=.true.,myrank=global%myrank)
1009 #endif
1010  do b=1,global%Nb
1011  call conservative2primitive(block = block(b))
1012  enddo
1013 #ifdef PROFILING
1014  call profile(p=2,pstop=.true.,myrank=global%myrank)
1015 #endif
1016 
1017  ! imposing the boundary conditions
1018 #ifdef PROFILING
1019  call profile(p=3,pstart=.true.,myrank=global%myrank)
1020 #endif
1021  call boundary_conditions(l = l, block = block)
1022 #ifdef PROFILING
1023  call profile(p=3,pstop=.true.,myrank=global%myrank)
1024 #endif
1025 
1026  ! saving the actual solution
1027  if (global%file%sol_out>0) then
1028  if ((mod(global%n,global%file%sol_out)==0).OR.(global%t==global%Tmax).OR.(global%n==global%Nmax)) then
1029  do b=1,global%Nb
1030  err= block(b)%save_fluid(filename=file_name(basename=trim(global%file%Path_OutPut)//global%file%File_Sol,&
1031  suffix=trim(global%file%Sol_Ext),blk=blockmap(b),grl=l,n=global%n))
1032  enddo
1033  endif
1034  endif
1035 
1036  ! saving the restart solution file
1037  if ((mod(global%n,global%file%restart_out)==0).OR.(global%t==global%Tmax).OR.(global%n==global%Nmax)) then
1038  flip = 1_i1p - flip
1039  do b=1,global%Nb
1040  err= block(b)%save_fluid(filename=file_name(basename=trim(global%file%Path_OutPut)//global%file%File_Sol,&
1041  suffix=trim(global%file%Sol_Ext),blk=blockmap(b),grl=l,flip=flip))
1042  enddo
1043  endif
1044 
1045  ! updating time varying variables: Dt,Dtmin
1046  global%n = global%n + 1_i8p
1047 #ifdef PROFILING
1048  call profile(p=4,pstart=.true.,myrank=global%myrank)
1049 #endif
1050  do b=1,global%Nb
1051  call compute_time(block=block(b),dtmin=dtmin(b))
1052  enddo
1053 #ifdef PROFILING
1054  call profile(p=4,pstop=.true.,myrank=global%myrank)
1055 #endif
1056  dtminl = minval(dtmin)
1057 #ifdef MPI2
1058  ! for multi-processes simulation all processes must exchange their DtminL for computing the global variables
1059  call mpi_allreduce(dtminl,gdtmin,1,mpi_real8,mpi_min,mpi_comm_world,err)
1060 #else
1061  ! for single processes DtminL are already global variables
1062  gdtmin = dtminl
1063 #endif
1064  if (global%unsteady) then ! for an unsteady accurate simulation each cell is updated by means of global minimum time step
1065  ! control for the last iterate
1066  if (global%Nmax<=0) then
1067  if ((global%t+gdtmin)>global%Tmax) then
1068  ! the global minimum time step is so high that the last iteration will go over Tmax
1069  ! it is decreased both for in order to achieve exactly Tmax
1070  gdtmin=abs(global%Tmax-global%t)
1071  endif
1072  endif
1073  global%t = global%t + gdtmin
1074  do b=1,global%Nb
1075  block(b)%C%Dt = gdtmin
1076  enddo
1077  endif
1078 
1079  ! updating console
1080  if (global%myrank==0) then
1081  if ((mod(global%n,global%file%screen_out)==0).OR.(global%t==global%Tmax).OR.(global%n==global%Nmax).OR.(global%n==1)) then
1082  sec_elp=crono(instant0=.true.)
1083  if (global%Nmax>0) then
1084  progress = global%n*100/(global%Nmax*1._r_p)
1085  sec_res = global%Nmax*sec_elp/global%n - sec_elp
1086  elseif (global%Tmax>0._r_p) then
1087  progress = 100*global%t/global%Tmax
1088  sec_res = global%Tmax*sec_elp/global%t - sec_elp
1089  else
1090  progress = 0._r_p
1091  sec_res = 0._r_p
1092  endif
1093  time_elp = seconds_to_time(sec_elp)
1094  time_res = seconds_to_time(sec_res)
1095  write(stdout,'(A)', iostat=err)'----------------------------------------------------------------------'
1096  write(stdout,'(A39,I30)', iostat=err)' Current grid level l: ',l
1097  write(stdout,'(A39,23X,F6.2,A)', iostat=err)' Simulation progress p: ',progress,'%'
1098  write(stdout,'(A39,I30)', iostat=err)' Current step number n: ',global%n
1099  write(stdout,'(A39,ES30.12)', iostat=err)' Current simulation time t: ',global%t
1100  write(stdout,'(A39,ES30.12)', iostat=err)' Current time step value gDt: ',gdtmin
1101  write(stdout,*)
1102  write(stdout,'(10X,A15,20X,A15)', iostat=err)' Elapsed Time','Residual Time'
1103  write(stdout,'(10X,A9,I6,20X,A9,I6)', iostat=err)' Days =',time_elp%Days, 'Days =',time_res%Days
1104  write(stdout,'(10X,A9,I6,20X,A9,I6)', iostat=err)' Hours =',time_elp%Hours, 'Hours =',time_res%Hours
1105  write(stdout,'(10X,A9,I6,20X,A9,I6)', iostat=err)' Minutes =',time_elp%Minutes,'Minutes =',time_res%Minutes
1106  write(stdout,'(10X,A9,F6.1,20X,A9,F6.1)',iostat=err)' Seconds =',time_elp%Seconds,'Seconds =',time_res%Seconds
1107  write(stdout,'(A)', iostat=err)'----------------------------------------------------------------------'
1108  write(stdout,*)
1109  endif
1110  endif
1111 
1112  ! evaluating the Runge-Kutta stages
1113  ! Runge-Kutta stages initialization
1114  do b=1,global%Nb

1149 #endif

1155 #endif
1156  ! finding the maximum value of residuals of actual process

1164 #endif

1168  if ((mod(global%n,global%file%probe_out)==0).OR.(global%t==global%Tmax).OR.(global%n==global%Nmax).OR.(global%n==1)) then
