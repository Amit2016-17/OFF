00490     ! Subroutine for computing interfaces fluxes. The 3D primitive variables are first projected in the 1D normal of the interface;

00492     ! 1D remapping:
00493     ! 1)    density of species 1    (r1)

00501     ! Ns+3) density                 (r=sum(rs))
00502     ! Ns+4) specific heats ratio    (g)
00503     !-------------------------------------------------------------------------------------------------------------------------------
00504 
00505     !-------------------------------------------------------------------------------------------------------------------------------
00506     implicit none
00507     integer(I1P),            intent(IN)::    gc                         ! Number of ghost cells used.
00508     integer(I_P),            intent(IN)::    N                          ! Number of cells.
00509     integer(I_P),            intent(IN)::    Ns                         ! Number of species.
00510     real(R_P),               intent(IN)::    cp0(1:Ns)                  ! Initial specific heat cp.

00963     pure subroutine positivity_preserving_limiter(o,vmean,vr)
00964     !-------------------------------------------------------------------------------------------------------------------------------
00965     ! Positivity preserving limiter.
00966     !-------------------------------------------------------------------------------------------------------------------------------
00967 
00968     !-------------------------------------------------------------------------------------------------------------------------------
00969     implicit none
00970     integer(I_P), intent(IN)::    o                                        ! Order of space reconstruction.
00971     real(R_P),    intent(IN)::    vmean                                    ! Mean value.
00972     real(R_P),    intent(INOUT):: vr(1:2)                                  ! Reconstructed values.
00973     real(R_P), parameter::        w1(2:4)=[1._R_P/6._R_P,                 ! Weights of Gauss-Lobatto's 3 points quadrature.
00974                                            1._R_P/12._R_P,                ! Weights of Gauss-Lobatto's 4 points quadrature.
00975                                            1._R_P/20._R_P]                 ! Weights of Gauss-Lobatto's 5 points quadrature.
00976     real(R_P), parameter::        w2(2:4)=[1._R_P/(1._R_P-2._R_P*w1(2)),  ! 1/(1-2*w1)
00977                                            1._R_P/(1._R_P-2._R_P*w1(3)),  ! 1/(1-2*w1)
00978                                            1._R_P/(1._R_P-2._R_P*w1(4))]   ! 1/(1-2*w1)
00979     real(R_P), parameter::        e  = tiny(1._R_P)                        ! Small number for avoid division for zero.
00980     real(R_P)::                   vstar                                    ! Star value.
00981     real(R_P)::                   theta                                    ! Limiter coefficient.
00982     !-------------------------------------------------------------------------------------------------------------------------------
00983 
00984     !-------------------------------------------------------------------------------------------------------------------------------
00985     vstar = (vmean - w1(o)*(vr(1) + vr(2)))*w2(o)
00986     theta = min((vmean-0.1_R_P*vmean)/(vmean - min(vstar,vr(1),vr(2)) + e),1._R_P)
00987     vr(1) = theta*(vr(1) - vmean) + vmean
00988     vr(2) = theta*(vr(2) - vmean) + vmean
00989     return

01413   subroutine solve_grl(myrank,l,global,block)
01414   !---------------------------------------------------------------------------------------------------------------------------------
01415   ! Subroutine for solving the conservation equations for a given grid level
01416   !---------------------------------------------------------------------------------------------------------------------------------
01417 
01418   !---------------------------------------------------------------------------------------------------------------------------------
01419   implicit none
01420   integer(I_P),      intent(IN)::    myrank                                   ! Actual rank process.
01421   integer(I_P),      intent(IN)::    l                                        ! Actual grid level.
01422   type(Type_Global), intent(INOUT):: global                                   ! Global-level data.
01423   type(Type_Block),  intent(INOUT):: block(1:global%mesh%Nb)                  ! Block-level data.
01424   real(R_P)::                        Dtmin(1:global%mesh%Nb)                  ! Min t step of actual process for each blk.

01430   integer(I_P)::                     err                                      ! Error trapping flag: 0 no errors, >0 errors.
01431   integer(I_P)::                     b                                        ! Blocks counter.
01432   integer(I_P)::                     s1                                       ! Runge-Kutta stages counters.
01433   real(R_P)::                        sec_elp                                  ! Seconds elapsed from the simulation start.
01434   real(R_P)::                        sec_res                                  ! Seconds residual from the simulation end.

01436   type(Type_Time)::                  time_res                                 ! Time residual (in days,hours,min... format).

01438   !---------------------------------------------------------------------------------------------------------------------------------
01439 
01440   !---------------------------------------------------------------------------------------------------------------------------------
01441   ! converting conservative variables to primitive ones

01443     call conservative2primitive(global = global, block = block(b))
01444   enddo
01445 
01446   ! imposing the boundary conditions
01447   call boundary_conditions(myrank = myrank, l = l, global = global, block = block)
01448 
01449   ! saving the actual solution
01450   if (global%file%sol_out>0) then
01451     if ((mod(global%fluid%n,global%file%sol_out)==0).OR. &
01452         (global%fluid%t==global%fluid%Tmax).OR.          &
01453         (global%fluid%n==global%fluid%Nmax)) then

01455         err=save_bfluid(filename=file_name(basename=trim(global%file%Path_OutPut)//global%file%File_Sol,&
01456                                            suffix='.sol',blk=blockmap(b),grl=l,n=global%fluid%n),       &
01457                         global=global,block=block(b))
01458       enddo
01459     endif
01460   endif
01461

01463   if ((mod(global%fluid%n,global%file%restart_out)==0).OR. &
01464       (global%fluid%t==global%fluid%Tmax).OR.              &
01465       (global%fluid%n==global%fluid%Nmax)) then

01468       err=save_bfluid(filename=file_name(basename=trim(global%file%Path_OutPut)//global%file%File_Sol,&
01469                                          suffix='.sol',blk=blockmap(b),grl=l,flip=flip),              &
01470                       global=global,block=block(b))

01472   endif

01477     call compute_time(global=global,block=block(b),Dtmin=Dtmin(b))

01494       endif
01495     endif

01526       write(stdout,'(A39,ES30.12)',            iostat=err)' Actual time step value           gDt: ',gDtmin
01527       write(stdout,*)

01530       write(stdout,'(10X,A9,I6,20X,A9,I6)',    iostat=err)' Hours   =',time_elp%Hours,  'Hours   =',time_res%Hours

01534       write(stdout,*)
01535     endif

01540   do b=1,global%mesh%Nb
01541     call init_cons(Ns=global%fluid%Ns,cons=block(b)%fluid%KS)
01542   enddo
01543   do s1=1,global%fluid%rk_ord
01544     if (s1>1) then
01545       ! summing the stages up to s1-1 for update P
01546       do b=1,global%mesh%Nb

01548       enddo
01549       ! imposing the boundary conditions
01550       call boundary_conditions(myrank = myrank, l = l, global = global, block = block)
01551     endif

01554       call residuals(s1=s1,global=global,block=block(b))
01555     enddo
01556   enddo
01557   ! Runge-Kutta time integration
01558   do b=1,global%mesh%Nb
01559     call rk_time_integration(global=global,block=block(b),RU=RU(:,b))
01560   enddo

01564   ! for multi-processes simulation all processes must exchange their mRU for computing the global gmRU

01574        (global%fluid%t==global%fluid%Tmax).OR. &
01575        (global%fluid%n==global%fluid%Nmax).OR. &
01576        (global%fluid%n==1)) then
01577       write(global%file%unit_res,trim(global%file%varform_res))global%fluid%n,global%fluid%t,(gmRU(b),b=1,global%fluid%Nc)
01578     endif
01579   endif
01580   return

01582   endsubroutine solve_grl

01653   subroutine rk_stages_sum(s1,global,block)
01654   !---------------------------------------------------------------------------------------------------------------------------------
01655   ! Subroutine for summing Runge-Kutta stages for updating primitive variables (block%fluid%P).

01666 
01667   !---------------------------------------------------------------------------------------------------------------------------------
01668   !$OMP PARALLEL DEFAULT(NONE)   &
01669   !$OMP PRIVATE(i,j,k,s,rksum,U) &
01670   !$OMP SHARED(s1,global,block,rk_c2)
01671   call init_cons(Ns = global%fluid%Ns, cons = rksum)
01672   call init_cons(Ns = global%fluid%Ns, cons = U    )

01675     do j=1,block%mesh%Nj

01680         enddo

01691   !---------------------------------------------------------------------------------------------------------------------------------
01692   ! Subroutine for computing Runge-Kutta integration.
01693   !---------------------------------------------------------------------------------------------------------------------------------
01694 
01695   !---------------------------------------------------------------------------------------------------------------------------------
01696   implicit none

01703 
01704   !---------------------------------------------------------------------------------------------------------------------------------

01706   !$OMP PARALLEL DEFAULT(NONE)     &
01707   !$OMP PRIVATE(i,j,k,s,rksum)     &
01708   !$OMP SHARED(global,block,rk_c1) &
01709   !$OMP REDUCTION(+: RU)
01710   call init_cons(Ns = global%fluid%Ns, cons = rksum)
01711   !$OMP DO
01712   do k=1,block%mesh%Nk
01713     do j=1,block%mesh%Nj
01714       do i=1,block%mesh%Ni

01716         do s=1,global%fluid%rk_ord

01718         enddo

01720         RU = RU + cons2array(rksum*rksum)

01722     enddo
