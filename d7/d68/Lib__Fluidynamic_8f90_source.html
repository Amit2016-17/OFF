00200         ! evaluating the maximum propagation speed of acoustic segnals multiplied for face area
00201         ! left i

00212         vmjL = abs(vmjL) + ss

00220         vmkL = abs(vmkL) + ss
00221         ! right k
00222         vm   = 0.5_R_P*(block%fluid%P(i,j,k)%v+block%fluid%P(i,j,k+1)%v)

00242       enddo
00243     enddo
00244   enddo
00245   ! right i frame

00247   do k=1-gc(5),Nk+gc(6)

00249       do i=Ni+1,Ni+gc(2)
00250         block%fluid%Dt(i,j,k) = Dtmin
00251       enddo
00252     enddo
00253   enddo
00254   ! left j frame
00255   !$OMP DO

00904   subroutine rk_time_integration(global,block,RU)
00905   !---------------------------------------------------------------------------------------------------------------------------------
00906   implicit none
00907   type(Type_Global), intent(IN)::    global                !< Global-level data.
00908   type(Type_Block),  intent(INOUT):: block                 !< Block-level data.
00909   real(R_P),         intent(OUT)::   RU(1:global%fluid%Nc) !< NormL2 of residuals of conservative variables.
00910   type(Type_Conservative)::          Ud,R                  !< Dummy conservative variables.
00911   integer(I8P)::                     i,j,k                 !< counters.
00912   !---------------------------------------------------------------------------------------------------------------------------------
00913 
00914   !---------------------------------------------------------------------------------------------------------------------------------
00915   RU = 0._R_P
00916   !$OMP PARALLEL DEFAULT(NONE) &
00917   !$OMP PRIVATE(i,j,k,Ud,R)    &
00918   !$OMP SHARED(global,block)   &
00919   !$OMP REDUCTION(+: RU)
00920   call Ud%init(Ns = global%fluid%Ns)
00921   call R%init( Ns = global%fluid%Ns)
00922   !$OMP DO
00923   do k=1,block%mesh%Nk
00924     do j=1,block%mesh%Nj
00925       do i=1,block%mesh%Ni
00926         call rk_time_integ(Dt=block%fluid%Dt(i,j,k),Un=block%fluid%U(i,j,k),KS=block%fluid%KS(i,j,k,1:global%fluid%rk_ord),Unp1=Ud)
00927         R%rs = (Ud%rs - block%fluid%U(i,j,k)%rs)/block%fluid%Dt(i,j,k) ; R%rs = R%rs*R%rs
00928         R%rv = (Ud%rv - block%fluid%U(i,j,k)%rv)/block%fluid%Dt(i,j,k) ; R%rv = R%rv*R%rv
00929         R%re = (Ud%re - block%fluid%U(i,j,k)%re)/block%fluid%Dt(i,j,k) ; R%re = R%re*R%re
00930         RU = RU + R%cons2array()
00931         block%fluid%U(i,j,k) = Ud
00932       enddo
00933     enddo
00934   enddo
00935   !$OMP DO
00936   do i=1,global%fluid%Nc
00937    RU(i) = sqrt(RU(i))
00938   enddo
00939   !$OMP END PARALLEL
00940   return
