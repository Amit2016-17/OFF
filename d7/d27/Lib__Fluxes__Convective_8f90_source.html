00185     pure subroutine preconstruct_n(gc,N,Ns,cp0,cv0,NF,P,PR)

00187     implicit none
00188     integer(I1P),         intent(IN)::  gc                                !< Number of ghost cells used.
00189     integer(I_P),         intent(IN)::  N                                 !< Number of cells.
00190     integer(I_P),         intent(IN)::  Ns                                !< Number of species.
00191     real(R_P),            intent(IN)::  cp0(1:Ns)                         !< Initial specific heat cp.
00192     real(R_P),            intent(IN)::  cv0(1:Ns)                         !< Initial specific heat cv.
00193     type(Type_Vector),    intent(IN)::  NF (                  0-gc: N+gc) !< Interface normal       [0-gc:N+gc].
00194     type(Type_Primitive), intent(IN)::  P  (                  1-gc: N+gc) !< 3D primitive variables [1-gc:N+gc].
00195     real(R_P),            intent(OUT):: PR (1:Ns+4,       1:2,   0: N+1 ) !< Reconstructed interface values 1D primitive variables.
00196     real(R_P)::                         P1D(1:Ns+2,       1:2,1-gc:-1+gc) !< 1D primitive variables at interfaces.
00197 #ifdef RECVC
00198     real(R_P)::                         Pm (1:Ns+4                      ) !< Mean of primitive variables across interfaces.
00199     real(R_P)::                         LPm(1:Ns+2,1:Ns+2,1:2           ) !< Mean left eigenvectors matrix.
00200     real(R_P)::                         RPm(1:Ns+2,1:Ns+2,1:2           ) !< Mean right eigenvectors matrix.
00201     real(R_P)::                         C  (1:Ns+2,       1:2,1-gc:-1+gc) !< Interface value of characteristic variables.
00202     real(R_P)::                         CR (1:Ns+2,       1:2           ) !< Reconstructed interface value of charac. variables.
00203 #endif
00204     logical::                           ROR(1:2)                          !< Logical flag for testing the result of WENO reconst.
00205     integer(I_P)::                      or                                !< Counter of order for ROR algorithm.
00206     integer(I_P)::                      i,j,k,v                           !< Counters.
00207     !-------------------------------------------------------------------------------------------------------------------------------
00208 
00209     !-------------------------------------------------------------------------------------------------------------------------------
00210     do i=0,N+1 ! loop over cells
00211       ! computing 1D primitive variables for the stencil [i+1-gc,i-1+gc] fixing the velocity across the interfaces i+-1/2
00212       do k=i+1-gc,i-1+gc
00213         do j=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00214           if (i==0  .AND.j==1) cycle
00215           if (i==N+1.AND.j==2) cycle
00216           P1D(1:Ns,j,k-i) =  P(k)%r(1:Ns)
00217           P1D(Ns+1,j,k-i) = (P(k)%v.dot.NF(i+j-2))
00218           P1D(Ns+2,j,k-i) =  P(k)%p
00219         enddo
00220       enddo
00221 #ifdef RECVC
00222       ! computing mean left and right eigenvectors matrix across the interface
00223       do j=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00224         if (i==0  .AND.j==1) cycle
00225         if (i==N+1.AND.j==2) cycle
00226         ! computing mean of primitive variables across the interface
00227         Pm(1:Ns) = 0.5_R_P*( P(i+j-1)%r(1:NS)          +  P(i+j-2)%r(1:NS)         ) ! rs
00228         Pm(Ns+1) = 0.5_R_P*((P(i+j-1)%v.dot.NF(i+j-2)) + (P(i+j-2)%v.dot.NF(i+j-2))) ! un
00229         Pm(Ns+2) = 0.5_R_P*( P(i+j-1)%p                +  P(i+j-2)%p               ) ! p
00230         Pm(Ns+3) = 0.5_R_P*( P(i+j-1)%d                +  P(i+j-2)%d               ) ! r
00231         Pm(Ns+4) = 0.5_R_P*( P(i+j-1)%g                +  P(i+j-2)%g               ) ! g
00232         ! computing mean left and right eigenvectors matrix across the interface
00233         LPm(1:Ns+2,1:Ns+2,j) = LP(Ns,Pm(1:Ns+4))
00234         RPm(1:Ns+2,1:Ns+2,j) = RP(Ns,Pm(1:Ns+4))
00235       enddo
00236       ! transforming variables into local characteristic fields for the stencil [1-gc,-1+gc]
00237       do k=i+1-gc,i-1+gc
00238         do j=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00239           if (i==0  .AND.j==1) cycle
00240           if (i==N+1.AND.j==2) cycle
00241           do v=1,Ns+2
00242             C(v,j,k-i) = dot_product(LPm(v,1:Ns+2,j),P1D(1:Ns+2,j,k-i))
00243           enddo
00244         enddo
00245       enddo
00246 #endif
00247       ! computing WENO reconstruction with ROR (Recursive Order Reduction) algorithm
00248       ROR_check: do or=gc,2,-1
00249         ! computing WENO reconstruction
00250 #ifdef RECVC
00251         do v=1,Ns+2
00252           !CR(v,1:2) = weno(S=or,V=C(v,1:2,1-or:-1+or))
00253           call weno(S=or,V=C(v,1:2,1-or:-1+or),VR=CR(v,1:2))
00254         enddo
00255         ! trasforming local reconstructed characteristic variables to primitive ones
00256         do j=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00257           if (i==0  .AND.j==1) cycle
00258           if (i==N+1.AND.j==2) cycle
00259           do v=1,Ns+2
00260             PR(v,j,i) = dot_product(RPm(v,1:Ns+2,j),CR(1:Ns+2,j))
00261           enddo
00262         enddo
00263 #else
00264         do v=1,Ns+2
00265           !PR(v,1:2,i) = weno(S=or,V=P1D(v,1:2,1-or:-1+or))
00266           call weno(S=or,V=P1D(v,1:2,1-or:-1+or),VR=PR(v,1:2,i))
00267         enddo
00268 #endif
00269         ! extrapolation of the reconstructed values for the non computed boundaries
00270         if (i==0  ) PR(1:Ns+2,1,i) = PR(1:Ns+2,2,i)
00271         if (i==N+1) PR(1:Ns+2,2,i) = PR(1:Ns+2,1,i)
00272 #ifdef PPL
00273         ! applaying the maximum-principle-satisfying limiter to the reconstructed densities and pressure
00274         do v=1,Ns
00275           call positivity_preserving_limiter(o = or, vmean = P(i)%r(v), vr = PR(v,1:2,i)) ! densities
00276         enddo
00277         call positivity_preserving_limiter(o = or, vmean = P(i)%p, vr = PR(Ns+2,1:2,i))   ! pressure
00278 #endif
00279         ! computing the reconstructed variables of total density and specific heats ratio
00280         do j=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00281           if (i==0  .AND.j==1) cycle
00282           if (i==N+1.AND.j==2) cycle
00283           PR(Ns+3,j,i) = sum(PR(1:Ns,j,i))
00284           PR(Ns+4,j,i) = dot_product(PR(1:Ns,j,i)/PR(Ns+3,j,i),cp0(1:Ns))/dot_product(PR(1:Ns,j,i)/PR(Ns+3,j,i),cv0(1:Ns))
00285         enddo
00286         ! verifing the WENO reconstruction at order or-th
00287         ROR = .true.
00288         do j=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00289           if (i==0  .AND.j==1) cycle
00290           if (i==N+1.AND.j==2) cycle
00291           ROR(j) = ((PR(Ns+2,j,i)>0._R_P).AND.(PR(Ns+3,j,i)>0._R_P).AND.(PR(Ns+4,j,i)>0._R_P))
00292         enddo
00293         if (ROR(1).AND.ROR(2)) exit ROR_check
00294         if (or==2) then
00295           ! the high order reconstruction is falied; used 1st order reconstruction
00296           do j=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00297             if (i==0  .AND.j==1) cycle
00298             if (i==N+1.AND.j==2) cycle
00299             PR(1:Ns,j,i) =  P(i)%r(1:Ns)          ! rs
00300             PR(Ns+1,j,i) = (P(i)%v.dot.NF(i+j-2)) ! un
00301             PR(Ns+2,j,i) =  P(i)%p                ! p
00302             PR(Ns+3,j,i) =  P(i)%d                ! r
00303             PR(Ns+4,j,i) =  P(i)%g                ! g
00304           enddo
00305         endif
00306       enddo ROR_check
00307     enddo

00319     pure function LP(Ns,primitive) result(L)
00320     !-------------------------------------------------------------------------------------------------------------------------------
00321     implicit none
00322     integer(I_P), intent(IN):: Ns                ! Number of species.
00323     real(R_P),    intent(IN):: primitive(1:Ns+4) ! Primitive variables.
00324     real(R_P)::                L(1:Ns+2,1:Ns+2)  ! Left eigenvectors matrix.
00325     real(R_P)::                gp                ! g*p.

00327     integer(I_P)::             i                 ! Counter.
00328     !-------------------------------------------------------------------------------------------------------------------------------
00329 
00330     !-------------------------------------------------------------------------------------------------------------------------------
00331     ! initializing
00332     gp = primitive(Ns+4)*primitive(Ns+2)
00333     a  = sqrt(gp/primitive(Ns+3))
00334     L  = 0._R_P
00335     ! assigning non-zero values of L

00360     real(R_P)::                a                 ! Speed of sound.
00361     integer(I_P)::             i                 ! Counter.
00362     !-------------------------------------------------------------------------------------------------------------------------------
00363 
00364     !-------------------------------------------------------------------------------------------------------------------------------
00365     ! initializing
00366     gp     = primitive(Ns+4)*primitive(Ns+2)
00367     a      = sqrt(gp/primitive(Ns+3))
00368     gp_inv = 1._R_P/gp
00369     R = 0._R_P
00370     ! assigning non-zero values of R

00373     enddo

00375       R(Ns+2,1) =  0.5_R_P          ;                                             R(Ns+2,Ns+2) = 0.5_R_P
00376     return

00382     pure subroutine positivity_preserving_limiter(o,vmean,vr)
00383     !-------------------------------------------------------------------------------------------------------------------------------
00384     implicit none
00385     integer(I_P), intent(IN)::    o                                        ! Order of space reconstruction.
00386     real(R_P),    intent(IN)::    vmean                                    ! Mean value.
00387     real(R_P),    intent(INOUT):: vr(1:2)                                  ! Reconstructed values.
00388     real(R_P), parameter::        w1(2:4)=[1._R_P/6._R_P,                 ! Weights of Gauss-Lobatto's 3 points quadrature.
00389                                            1._R_P/12._R_P,                ! Weights of Gauss-Lobatto's 4 points quadrature.
00390                                            1._R_P/20._R_P]                 ! Weights of Gauss-Lobatto's 5 points quadrature.
00391     real(R_P), parameter::        w2(2:4)=[1._R_P/(1._R_P-2._R_P*w1(2)),  ! 1/(1-2*w1)
00392                                            1._R_P/(1._R_P-2._R_P*w1(3)),  ! 1/(1-2*w1)
00393                                            1._R_P/(1._R_P-2._R_P*w1(4))]   ! 1/(1-2*w1)
00394     real(R_P), parameter::        e  = tiny(1._R_P)                        ! Small number for avoid division for zero.
00395     real(R_P)::                   vstar                                    ! Star value.
00396     real(R_P)::                   theta                                    ! Limiter coefficient.
00397     !-------------------------------------------------------------------------------------------------------------------------------
00398 
00399     !-------------------------------------------------------------------------------------------------------------------------------
00400     vstar = (vmean - w1(o)*(vr(1) + vr(2)))*w2(o)
00401     theta = min((vmean-0.1_R_P*vmean)/(vmean - min(vstar,vr(1),vr(2)) + e),1._R_P)
00402     vr(1) = theta*(vr(1) - vmean) + vmean
00403     vr(2) = theta*(vr(2) - vmean) + vmean
00404     return
