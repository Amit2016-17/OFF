146  subroutine parsing_command_line()
147  !---------------------------------------------------------------------------------------------------------------------------------
148  ! Subroutine for printing the correct use of the program.
149  !---------------------------------------------------------------------------------------------------------------------------------
150 
151  !---------------------------------------------------------------------------------------------------------------------------------
152  implicit none
153  integer(I_P):: nca = 0 ! Number of command line arguments.
154  integer(I_P):: c ! Counter for command line arguments.
155  character(6):: ca_switch ! Switch identifier.
156  character(3):: yes ! Yes (no) flag.
157  character(3):: os_type = 'UIX' ! OS type 'UIX' or 'WIN'.
158  character(20):: date ! Actual date.
159  !---------------------------------------------------------------------------------------------------------------------------------
160 
161  !---------------------------------------------------------------------------------------------------------------------------------
162  ! initializing solution file name to "unset" in order to capture when only mesh is to be post-processed
163  global%file%File_Sol = "unset"
164  nca = command_argument_count()
165  if (nca==0) then
166  write(stderr,'(A)')' Error: no argument has been passed to command line'
167  call print_usage
168  stop
169  else
170  ! processing switch
171  c = 0
172  do while (c<nca)
173  c = c + 1
174  call get_command_argument(c, ca_switch)
175  select case(adjustl(trim(ca_switch)))
176  case('-m') ! file name of mesh file
177  call get_command_argument(c+1, global%file%File_Mesh) ; c = c + 1
178  case('-o') ! file name of output file
179  call get_command_argument(c+1, global%file%File_Pout) ; c = c + 1
180  case('-s') ! file name of solution file
181  call get_command_argument(c+1, global%file%File_Sol) ; c = c + 1
182  case('-bc') ! saving boundary conditions ghost cells
183  pp_format%bc = .true.
184  case('-cell') ! dependent variables are saved cell centered instead of interpolated at nodes
185  pp_format%node = .false.
186  case('-ascii') ! ascii output
187  pp_format%binary = .false.
188  case('-tec') ! Tecplot file format
189  call get_command_argument(c+1,yes) ; c = c + 1 ; yes = upper_case(adjustl(trim(yes)))
190  pp_format%tec = (adjustl(trim(yes))=='YES')
191  case('-vtk') ! VKT file format
192  call get_command_argument(c+1,yes) ; c = c + 1 ; yes = upper_case(adjustl(trim(yes)))
193  pp_format%vtk = (adjustl(trim(yes))=='YES')
194  case('-gnu') ! Gnuplot file format
195  call get_command_argument(c+1,yes) ; c = c + 1 ; yes = upper_case(adjustl(trim(yes)))
196  pp_format%gnu = (adjustl(trim(yes))=='YES')
197  case('-os') ! OS type
198  call get_command_argument(c+1,os_type) ; c = c + 1 ; os_type = upper_case(os_type)
199  case default

202  stop
