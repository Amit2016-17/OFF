740  pure subroutine encode_bits(bits,padd,code)
741  !---------------------------------------------------------------------------------------------------------------------------------
742  implicit none
743  integer(I1P), intent(IN):: bits(1:)
744  integer(I4P), intent(IN):: padd
745  character(1), intent(OUT):: code(1:)
746  integer(I1P):: sixb(1:4)
747  integer(I8P):: c,e
748  !---------------------------------------------------------------------------------------------------------------------------------
749 
750  !---------------------------------------------------------------------------------------------------------------------------------
751  c = 1_i8p
752  do e=1_i8p,size(bits,dim=1),3_i8p ! loop over array elements: 3 bytes (24 bits) scanning
753  sixb = 0_i1p
754  call mvbits(bits(e ),2,6,sixb(1),0)
755  call mvbits(bits(e ),0,2,sixb(2),4) ; call mvbits(bits(e+1),4,4,sixb(2),0)
756  call mvbits(bits(e+1),0,4,sixb(3),2) ; call mvbits(bits(e+2),6,2,sixb(3),0)
757  call mvbits(bits(e+2),0,6,sixb(4),0)
758  sixb = sixb + 1_i1p
759  code(c :c )(1:1) = base64(sixb(1):sixb(1))
760  code(c+1:c+1)(1:1) = base64(sixb(2):sixb(2))
761  code(c+2:c+2)(1:1) = base64(sixb(3):sixb(3))
762  code(c+3:c+3)(1:1) = base64(sixb(4):sixb(4))
763  c = c + 4_i8p
764  enddo
765  if (padd>0) code(size(code,dim=1)-padd+1:)(1:1)='='
766  return
767  !---------------------------------------------------------------------------------------------------------------------------------
768  endsubroutine encode_bits
