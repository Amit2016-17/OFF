204  subroutine compute_values0(global)
205  !---------------------------------------------------------------------------------------------------------------------------------
206  implicit none
207  type(type_global), intent(INOUT):: global
208  !---------------------------------------------------------------------------------------------------------------------------------
209 
210  !---------------------------------------------------------------------------------------------------------------------------------
211  ! reference values
212  global%adim%mu0 = (global%adim%r0*global%adim%v0*global%adim%L0)/global%adim%Re
213  global%adim%f0 = (global%adim%v0*global%adim%v0)/(global%adim%L0*global%adim%Fr*global%adim%Fr)
214  global%adim%k0 = (global%adim%mu0*global%adim%c0)/global%adim%Pr
215  global%adim%Dt0 = global%adim%L0/global%adim%v0
216  global%adim%p0 = global%adim%r0*global%adim%v0*global%adim%v0
217  global%adim%a0 = global%adim%v0
218  global%adim%T0 = (global%adim%v0*global%adim%v0)/global%adim%c0
219  global%adim%E0 = global%adim%v0*global%adim%v0
220  global%adim%q0 = (global%adim%v0*global%adim%v0*global%adim%v0)/global%adim%L0
221  ! equations coefficients
222  global%adim%Re_inv = 1._r_p/global%adim%Re
223  global%adim%Fr2_inv = 1._r_p/(global%adim%Fr*global%adim%Fr)
224  global%adim%PrRe_inv = 1._r_p/(global%adim%Pr*global%adim%Re)
225  return
226  !---------------------------------------------------------------------------------------------------------------------------------
227  endsubroutine compute_values0

372  function load_gfluid_0species(global,filename) result(err)
373  !---------------------------------------------------------------------------------------------------------------------------------

383  !---------------------------------------------------------------------------------------------------------------------------------

385  !---------------------------------------------------------------------------------------------------------------------------------

388  open(unit = get_unit(unitfree), file = adjustl(trim(filename)), status = 'OLD', action = 'READ', form = 'FORMATTED')
389  read(unitfree,*,iostat=err) ! record skipped
390  read(unitfree,*,iostat=err) ! record skipped
391  read(unitfree,*,iostat=err)global%Ns ; call global%alloc_fluid
392  do s=1,global%Ns
393  read(unitfree,*,iostat=err) ! record skipped
394  read(unitfree,*,iostat=err)g
395  read(unitfree,*,iostat=err)r
396  read(unitfree,*,iostat=err)global%cp0(s)
397  read(unitfree,*,iostat=err)global%cv0(s)
398  if ((g>0.0_r_p).AND.(r>0.0_r_p)) then
399  global%cv0(s)=r/(g-1._r_p)
400  global%cp0(s)=g*global%cv0(s)
401  elseif ((g>0.0_r_p).AND.(global%cp0(s)>0.0_r_p)) then
402  global%cv0(s)=global%cp0(s)/g
403  elseif ((g>0.0_r_p).AND.(global%cv0(s)>0.0_r_p)) then
404  global%cp0(s)=global%cv0(s)*g
405  elseif ((r>0.0_r_p).AND.(global%cp0(s)>0.0_r_p)) then
406  global%cv0(s)=global%cp0(s)-r
407  elseif ((r>0.0_r_p).AND.(global%cv0(s)>0.0_r_p)) then
408  global%cp0(s)=global%cv0(s)+r
409  endif
410  enddo
411  close(unitfree)
412  return

414  endfunction load_gfluid_0species
