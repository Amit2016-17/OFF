00395   pure function weno_weights(S,V) result(w)
00396   !---------------------------------------------------------------------------------------------------------------------------------
00397   ! Function for computing WENO weights of the polynomial reconstructions.
00398   !---------------------------------------------------------------------------------------------------------------------------------
00399 
00400   !---------------------------------------------------------------------------------------------------------------------------------
00401   implicit none
00402   integer(I_P), intent(IN):: S                      ! Number of stencils used.
00403   real(R_P),    intent(IN):: V    (1:2,1-S:-1+S)    ! Variable to be reconstructed.
00404   real(R_P)::                w    (1:2,0:S-1)       ! Weights of the stencils.
00405   real(R_P)::                IS   (1:2,0:S-1)       ! Smoothness indicators of the stencils.
00406   real(R_P)::                a    (1:2,0:S-1)       ! Alpha coifficients for the weights.
00407   real(R_P)::                a_tot(1:2)             ! Summ of the alpha coefficients.
00408 #ifdef WENOZ
00409   real(R_P)::                tao  (1:2)             ! Normalization factor.
00410 #endif
00411   integer(I_P)::             s1,s2,s3,f             ! Counters.
00412   !---------------------------------------------------------------------------------------------------------------------------------
00413 
00414   !---------------------------------------------------------------------------------------------------------------------------------
00415   ! computing smoothness indicators
00416   do s1=0,S-1 ! stencil counter
00417     do f=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00418       IS(f,s1) = 0._R_P
00419       do s2=0,S-1
00420         do s3=0,S-1
00421           IS(f,s1) = IS(f,s1) + weno_d(s3,s2,s1)*V(f,s1-s3)*V(f,s1-s2)
00422         enddo
00423       enddo
00424     enddo
00425   enddo
00426 #ifdef WENOZ
00427   ! computing smooth-difference tao
00428   if (S>2) then
00429     do f=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00430       tao(f) = abs(IS(f,0) - (1-weno_odd)*IS(f,1) - (1-weno_odd)*IS(f,S-2) + (1-2*weno_odd)*IS(f,S-1))
00431     enddo
00432   else
00433     do f=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00434       tao(f) = udiff(p=2*S-2,v=V(f,1-S:S-1))
00435       tao(f) = tao(f)*tao(f)
00436     enddo
00437   endif
00438 #endif
00439   ! computing alfa coefficients
00440   a_tot = 0._R_P
00441   do s1=0,S-1
00442     do f=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00443 #ifdef WENOZ
00444       if (tao(f)>minval(IS(f,:))) then
00445         a(f,s1) = weno_a(f,s1)*(1._R_P+(tao(f)/(weno_eps+IS(f,s1)))**(weno_exp)) ; a_tot(f) = a_tot(f) + a(f,s1)
00446       else
00447         a(f,s1) = weno_a(f,s1)                                                   ; a_tot(f) = a_tot(f) + a(f,s1)
00448       endif
00449 #else
00450       a(f,s1) = weno_a(f,s1)*(1._R_P/(weno_eps+IS(f,s1))**S)                     ; a_tot(f) = a_tot(f) + a(f,s1)
00451 #endif
00452     enddo
00453   enddo
00454   ! computing the weights
00455   do s1=0,S-1
00456     do f=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00457       w(f,s1) = a(f,s1)/a_tot(f)
00458     enddo
00459   enddo
00460 #ifdef WENOM
00461   ! Henrick mapping
00462   a_tot = 0._R_P
00463   do s1=0,S-1
00464     do f=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00465       a(f,s1)  = ( w(f,s1)*( weno_a(f,s1) + weno_a(f,s1)*weno_a(f,s1) - 3._R_P*weno_a(f,s1)*w(f,s1) + w(f,s1)*w(f,s1) ) )/ &
00466                  ( weno_a(f,s1)*weno_a(f,s1) + w(f,s1)*( 1._R_P - 2._R_P*weno_a(f,s1) ) )
00467       a_tot(f) = a_tot(f) + a(f,s1)
00468     enddo
00469   enddo
00470   ! computing the weights with the mapped a(f,s1)
00471   do s1=0,S-1
00472     do f=1,2 ! 1 => left interface (i-1/2), 2 => right interface (i+1/2)
00473       w(f,s1) = a(f,s1)/a_tot(f)
00474     enddo
00475   enddo
00476 #endif
00477   return
