00282   subroutine compute_values0(global)
00283   !---------------------------------------------------------------------------------------------------------------------------------
00284   implicit none
00285   type(Type_Global), intent(INOUT):: global !< Global level data.
00286   !---------------------------------------------------------------------------------------------------------------------------------
00287 
00288   !---------------------------------------------------------------------------------------------------------------------------------
00289   ! reference values
00290   global%adim%mu0  = (global%adim%r0*global%adim%v0*global%adim%L0)/global%adim%Re
00291   global%adim%f0   = (global%adim%v0*global%adim%v0)/(global%adim%L0*global%adim%Fr*global%adim%Fr)
00292   global%adim%k0   = (global%adim%mu0*global%adim%c0)/global%adim%Pr
00293   global%adim%Dt0  = global%adim%L0/global%adim%v0
00294   global%adim%p0   = global%adim%r0*global%adim%v0*global%adim%v0
00295   global%adim%a0   = global%adim%v0
00296   global%adim%T0   = (global%adim%v0*global%adim%v0)/global%adim%c0
00297   global%adim%E0   = global%adim%v0*global%adim%v0
00298   global%adim%q0   = (global%adim%v0*global%adim%v0*global%adim%v0)/global%adim%L0
00299   ! equations coefficients
00300   global%adim%Re_inv   = 1._R_P/global%adim%Re
00301   global%adim%Fr2_inv  = 1._R_P/(global%adim%Fr*global%adim%Fr)
00302   global%adim%PrRe_inv = 1._R_P/(global%adim%Pr*global%adim%Re)
