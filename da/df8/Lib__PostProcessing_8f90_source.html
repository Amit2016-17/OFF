68  !---------------------------------------------------------------------------------------------------------------------------------
69  ! Subroutine for interpolating primitive variables at nodes.
70  !---------------------------------------------------------------------------------------------------------------------------------
71 
72  !---------------------------------------------------------------------------------------------------------------------------------
73  implicit none
74  type(type_sblock), intent(INOUT):: block ! Block-level data.
75  type(type_primitive), intent(INOUT):: p(0:block%ni,0:block%nj,0:block%nk) ! Primitive variables intepolated at nodes.
76  real(R_P):: mf ! Mean factor.
77  integer(I_P):: i,j,k ! Counters.
78  !---------------------------------------------------------------------------------------------------------------------------------
79 
80  !---------------------------------------------------------------------------------------------------------------------------------
81  call p%init(ns=block%global%Ns)
82 #if !defined NULi && !defined NULj && !defined NULk
83  ! 3D data
84  mf = 0.125_r_p
85 #elif defined NULi
86  block%C(0 ,:,:)%P = 0._r_p
87  block%C(block%Ni+1,:,:)%P = 0._r_p
88 #if !defined NULj && !defined NULk
89  ! 2D data
90  mf = 0.25_r_p
91 #elif defined NULj
92  ! 1D data
93  mf = 0.5_r_p
94  block%C(:,0 ,:)%P = 0._r_p
95  block%C(:,block%Nj+1,:)%P = 0._r_p
96 #elif defined NULk
97  ! 1D data
98  mf = 0.5_r_p
99  block%C(:,:,0 )%P = 0._r_p
100  block%C(:,:,block%Nk+1)%P = 0._r_p
101 #endif
102 #elif defined NULj
103  block%C(:,0 ,:)%P = 0._r_p
104  block%C(:,block%Nj+1,:)%P = 0._r_p
105 #if !defined NULi && !defined NULk
106  ! 2D data
107  mf = 0.25_r_p
108 #elif defined NULi
109  ! 1D data
110  mf = 0.5_r_p
111  block%C(0 ,:,:)%P = 0._r_p
112  block%C(block%Ni+1,:,:)%P = 0._r_p
113 #elif defined NULk
114  ! 1D data
115  mf = 0.5_r_p
116  block%C(:,:,0 )%P = 0._r_p
117  block%C(:,:,block%Nk+1)%P = 0._r_p
118 #endif
119 #elif defined NULk
120  block%C(:,:,0 )%P = 0._r_p
121  block%C(:,:,block%Nk+1)%P = 0._r_p
122 #if !defined NULi && !defined NULj
123  ! 2D data
124  mf = 0.25_r_p
125 #elif defined NULi
126  ! 1D data
127  mf = 0.5_r_p
128  block%C(0 ,:,:)%P = 0._r_p
129  block%C(block%Ni+1,:,:)%P = 0._r_p
130 #elif defined NULj
131  ! 1D data
132  mf = 0.5_r_p
133  block%C(:,0 ,:)%P = 0._r_p
134  block%C(:,block%Nj+1,:)%P = 0._r_p
135 #endif
136 #endif
137  !$OMP PARALLEL DEFAULT(NONE) &
138  !$OMP PRIVATE(i,j,k) &
139  !$OMP SHARED(block,P,mf)
140  !$OMP DO
141  do k=0,block%Nk
142  do j=0,block%Nj
143  do i=0,block%Ni
144  p(i,j,k) = block%C(i+1,j+1,k+1)%P + block%C(i,j+1,k+1)%P &
145  + block%C(i+1,j ,k+1)%P + block%C(i,j, k+1)%P &
146  + block%C(i+1,j+1,k )%P + block%C(i,j+1,k )%P &
147  + block%C(i+1,j ,k )%P + block%C(i,j ,k )%P

150  enddo
151  enddo
152  !$OMP END PARALLEL

157  subroutine compute_dimensions(block,ni1,ni2,nj1,nj2,nk1,nk2,ci1,ci2,cj1,cj2,ck1,ck2)

177  ck1 = 0 ; ck2 = block%Nk

186  ck1 = 1 - block%gc(5) ; ck2 = block%Nk + block%gc(6)
187  else
188  ni1 = 0 ; ni2 = block%Ni
189  nj1 = 0 ; nj2 = block%Nj
190  nk1 = 0 ; nk2 = block%Nk
191 
192  ci1 = 1 ; ci2 = block%Ni
193  cj1 = 1 ; cj2 = block%Nj
194  ck1 = 1 ; ck2 = block%Nk
195  endif
196  endif
197  return

354  integer(I_P):: err ! Error trapping flag: 0 no errors, >0 error occurs.

383  enddo
384  enddo

546  function vtk_blk_data(global,block,filename) result(err)
547  !-------------------------------------------------------------------------------------------------------------------------------
548  ! Function for writing block data.
549  !-------------------------------------------------------------------------------------------------------------------------------
550 
551  !-------------------------------------------------------------------------------------------------------------------------------
552  implicit none
553  type(type_global), intent(IN):: global ! Global-level data.
554  type(type_sblock), intent(INOUT):: block ! Block-level data.
555  character(*), intent(IN):: filename ! File name of the output block file.
556  integer(I_P):: err ! Error trapping flag: 0 no errors, >0 error occurs.
557  type(type_primitive):: p( 0:block%ni,0:block%nj,0:block%nk) ! Primitive variables intepolated at nodes.
558  real(R_P):: r(1:global%ns,0:block%ni,0:block%nj,0:block%nk) ! Array containing species densities.
559  integer(I_P):: ni1,ni2,nj1,nj2,nk1,nk2 ! Bounds of dimensions of node-centered data.
560  integer(I_P):: ci1,ci2,cj1,cj2,ck1,ck2 ! Bounds of dimensions of cell-centered data.
561  integer(I_P):: nnode,ncell ! Number of nodes and cells.
562  integer(I_P):: i,j,k,s ! Counters.
563  !-------------------------------------------------------------------------------------------------------------------------------
564 
565  !-------------------------------------------------------------------------------------------------------------------------------
566  if (pp_format%node) then
567  ! tri-linear interpolation of cell-centered values at nodes
568  call interpolate_primitive(block=block,p=p)
569  endif
570  ! initialize the block dimensions
571  call compute_dimensions(block=block, &
572  ni1=ni1,ni2=ni2,nj1=nj1,nj2=nj2,nk1=nk1,nk2=nk2, &
573  ci1=ci1,ci2=ci2,cj1=cj1,cj2=cj2,ck1=ck1,ck2=ck2)
574  nnode = (ni2-ni1+1)*(nj2-nj1+1)*(nk2-nk1+1)
575  ncell = (ci2-ci1+1)*(cj2-cj1+1)*(ck2-ck1+1)
576  ! storing species densities into an array for avoiding problems with nonzero rank pointer
577  if (pp_format%node) then
578  do k=0,block%Nk
579  do j=0,block%Nj
580  do i=0,block%Ni
581  do s=1,global%Ns
582  r(s,i,j,k) = p(i,j,k)%r(s)
583  enddo
584  enddo
585  enddo
586  enddo
587  else
588  do k=1,block%Nk
589  do j=1,block%Nj
590  do i=1,block%Ni
591  do s=1,global%Ns
592  r(s,i,j,k) = block%C(i,j,k)%P%r(s)
593  enddo
594  enddo
595  enddo
596  enddo
597  endif
598  ! initializing VTK file
599  if (pp_format%binary) then
600  err = vtk_ini_xml(output_format = 'binary', &
601  filename = trim(filename), &
602  mesh_topology = 'StructuredGrid', &
603  nx1 = ni1, nx2 = ni2, &
604  ny1 = nj1, ny2 = nj2, &
605  nz1 = nk1, nz2 = nk2)
606  else
607  err = vtk_ini_xml(output_format = 'ascii', &
608  filename = trim(filename), &
609  mesh_topology = 'StructuredGrid', &
610  nx1 = ni1, nx2 = ni2, &
611  ny1 = nj1, ny2 = nj2, &
612  nz1 = nk1, nz2 = nk2)
613  endif
614  ! saving auxiliary data (time and time step)
615  err = vtk_fld_xml(fld_action='open')
616  err = vtk_fld_xml(fld=block%global%t,fname='TIME')
617  err = vtk_fld_xml(fld=block%global%n,fname='CYCLE')
618  err = vtk_fld_xml(fld_action='close')
619  ! saving the geometry
620  err = vtk_geo_xml(nx1 = ni1, nx2 = ni2, &
621  ny1 = nj1, ny2 = nj2, &
622  nz1 = nk1, nz2 = nk2, &
623  nn = nnode, &
624  x=block%node(ni1:ni2,nj1:nj2,nk1:nk2)%x, &
625  y=block%node(ni1:ni2,nj1:nj2,nk1:nk2)%y, &
626  z=block%node(ni1:ni2,nj1:nj2,nk1:nk2)%z)
627  if (.not.meshonly) then
628  ! saving dependent variables
629  if (pp_format%node) then
630  err = vtk_dat_xml(var_location = 'node', var_block_action = 'open')
631  do s=1,global%Ns
632  err=vtk_var_xml(nc_nn=nnode,varname='r('//trim(str(.true.,s))//')',var=r(s,ni1:ni2,nj1:nj2,nk1:nk2))
633  enddo
634  err=vtk_var_xml(nc_nn=nnode,varname='r',var=p(ni1:ni2,nj1:nj2,nk1:nk2)%d )
635  err=vtk_var_xml(nc_nn=nnode,varname='u',var=p(ni1:ni2,nj1:nj2,nk1:nk2)%v%x)
636  err=vtk_var_xml(nc_nn=nnode,varname='v',var=p(ni1:ni2,nj1:nj2,nk1:nk2)%v%y)
637  err=vtk_var_xml(nc_nn=nnode,varname='w',var=p(ni1:ni2,nj1:nj2,nk1:nk2)%v%z)
638  err=vtk_var_xml(nc_nn=nnode,varname='p',var=p(ni1:ni2,nj1:nj2,nk1:nk2)%p )
639  err=vtk_var_xml(nc_nn=nnode,varname='g',var=p(ni1:ni2,nj1:nj2,nk1:nk2)%g )
640  err=vtk_dat_xml(var_location ='node',var_block_action = 'close')
641  else
642  err = vtk_dat_xml(var_location = 'cell', var_block_action = 'open')
643  do s=1,global%Ns
644  err=vtk_var_xml(nc_nn=ncell,varname='r('//trim(str(.true.,s))//')',var=r(s,ci1:ci2,cj1:cj2,ck1:ck2))
645  enddo
646  err=vtk_var_xml(nc_nn=ncell,varname='r',var=block%C(ci1:ci2,cj1:cj2,ck1:ck2)%P%d )
647  err=vtk_var_xml(nc_nn=ncell,varname='u',var=block%C(ci1:ci2,cj1:cj2,ck1:ck2)%P%v%x)
648  err=vtk_var_xml(nc_nn=ncell,varname='v',var=block%C(ci1:ci2,cj1:cj2,ck1:ck2)%P%v%y)
649  err=vtk_var_xml(nc_nn=ncell,varname='w',var=block%C(ci1:ci2,cj1:cj2,ck1:ck2)%P%v%z)
650  err=vtk_var_xml(nc_nn=ncell,varname='p',var=block%C(ci1:ci2,cj1:cj2,ck1:ck2)%P%p )
651  err=vtk_var_xml(nc_nn=ncell,varname='g',var=block%C(ci1:ci2,cj1:cj2,ck1:ck2)%P%g )
652  err=vtk_dat_xml(var_location ='cell',var_block_action = 'close')
653  endif
654  endif
655  ! closing VTK file
656  err = vtk_geo_xml()
657  err = vtk_end_xml()
658  return

660  endfunction vtk_blk_data
